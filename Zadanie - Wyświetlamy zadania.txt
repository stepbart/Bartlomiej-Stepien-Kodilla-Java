SELECT I.SUMMARY AS ISSUE_NAME, IL.NAME
FROM ISSUES I JOIN ISSUESLISTS IL on I.ISSUESLISTS_ID = IL.ID
GROUP BY I.ID
ORDER BY IL.ID, I.ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY U.LASTNAME, U.FIRSTNAME;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS NUMBER_OF_ISSUES
FROM ISSUES I JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*)>=1
ORDER BY I.USER_ID_ASSIGNEDTO DESC;
